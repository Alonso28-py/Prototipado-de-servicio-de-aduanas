    <script>
        function simularEnvio() {
            // Iteración 2.1: Validaciones y Reglas de Negocio Robustas
            const btn = document.querySelector('button[type="submit"]');
            const resultado = document.getElementById('resultado');
            const mensajeError = document.getElementById('mensajeError');
            
            // Obtener valores de los campos
            const rut = document.getElementById('rut').value.trim();
            const patente = document.getElementById('patente').value.trim().toUpperCase();

            // Ocultar mensajes previos
            resultado.classList.add('hidden', 'scale-95', 'opacity-0');
            mensajeError.classList.add('hidden');

            // Validación de Usabilidad (Evita enviar si hay campos vacíos)
            if (rut === '' || patente === '') {
                mensajeError.classList.remove('hidden');
                return; // Corta la ejecución de la función aquí mismo
            }

            // Regla de Negocio: Días de vigencia
            // Limpiamos la patente de puntos y espacios para una validación más exacta
            const patenteLimpia = patente.replace(/[\.\- ]/g, '');
            
            let diasOtorgados = 180;
            // Si la patente empieza con CD o CC (Cuerpo Diplomático/Consular), son 90 días
            if (patenteLimpia.startsWith('CD') || patenteLimpia.startsWith('CC')) {
                diasOtorgados = 90;
            }

            // Efecto visual de carga simple
            btn.innerHTML = '<svg class="animate-spin h-5 w-5 mr-3 text-white inline-block" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4" fill="none"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg> Procesando...';
            btn.disabled = true;
            btn.classList.add('opacity-70', 'cursor-not-allowed');

            // Ocultar mensaje previo si existiera
            resultado.classList.add('hidden', 'scale-95', 'opacity-0');

            setTimeout(() => {
                // Restaurar botón
                btn.innerHTML = 'Generar Trámite';
                btn.disabled = false;
                btn.classList.remove('opacity-70', 'cursor-not-allowed');
                
                // Actualizar texto de respuesta con la patente y los días otorgados
                document.getElementById('textoPatente').innerText = `Vehículo patente: ${patente}`;
                document.getElementById('textoDias').innerText = `${diasOtorgados} DÍAS OTORGADOS`;

                // Mostrar resultado con animación
                resultado.classList.remove('hidden');
                // Pequeño delay para que la transición CSS surta efecto
                setTimeout(() => {
                    resultado.classList.remove('scale-95', 'opacity-0');
                    resultado.classList.add('scale-100', 'opacity-100');
                }, 10);
            }, 800);
        }
    </script>

    <script>
        function procesarTramite() {
            const btn = document.getElementById('btnGenerar');
            const resultado = document.getElementById('resultado');
            const mensajeError = document.getElementById('mensajeError');
            const qrContainer = document.getElementById('qrcode');
            
            // 1. Obtener valores
            const rut = document.getElementById('rut').value.trim();
            const patente = document.getElementById('patente').value.trim().toUpperCase();

            // 2. Limpiar caracteres para validación estricta (Elimina puntos, guiones y espacios)
            const rutLimpio = rut.replace(/[^0-9Kk]/g, ''); 
            const patenteLimpia = patente.replace(/[^A-Z0-9]/g, ''); 

            // Reiniciar estados visuales
            resultado.classList.add('hidden', 'scale-95', 'opacity-0');
            mensajeError.classList.add('hidden');
            qrContainer.innerHTML = ''; 

            // === REGLAS DE VALIDACIÓN ESTRICTA ===

            // Regla A: Campos Vacíos
            if (rutLimpio === '' || patenteLimpia === '') {
                mostrarError('Por favor, complete tanto el RUT como la Patente para continuar.');
                return;
            }

            // Regla B: Longitud mínima del RUT (Asumimos mínimo 8 caracteres ej: 12345678)
            if (rutLimpio.length < 8) {
                mostrarError(`El RUT ingresado ("${rut}") está incompleto. Faltan números.`);
                return;
            }

            // Regla C: Longitud de Patente
            const esDiplomatica = patenteLimpia.startsWith('CD') || patenteLimpia.startsWith('CC');
            
            if (!esDiplomatica && patenteLimpia.length < 5) {
                mostrarError(`La patente estándar ("${patente}") es muy corta. Debe tener al menos 5 o 6 caracteres.`);
                return;
            }

            if (esDiplomatica && patenteLimpia.length < 2) {
                mostrarError('La patente diplomática está incompleta.');
                return;
            }

            // === REGLA DE NEGOCIO (DÍAS) ===
            let diasOtorgados = 180;
            if (esDiplomatica) {
                diasOtorgados = 90;
            }

            // Efecto de carga en el botón
            const textoOriginalBtn = btn.innerHTML;
            btn.innerHTML = '<svg class="animate-spin h-5 w-5 text-white" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4" fill="none"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg>';
            btn.disabled = true;

            // Simular tiempo de respuesta del servidor (800ms)
            setTimeout(() => {
                // Restaurar botón
                btn.innerHTML = textoOriginalBtn;
                btn.disabled = false;
                
                // Actualizar Textos
                document.getElementById('textoPatente').innerText = patenteLimpia; // Mostramos la patente limpia
                document.getElementById('textoDias').innerText = `${diasOtorgados} DÍAS`;

                // === GENERACIÓN DEL CÓDIGO QR ===
                const fechaActual = new Date().toLocaleDateString('es-CL');
                const datosQR = `SGAI | APROBADO | RUT:${rutLimpio} | PATENTE:${patenteLimpia} | PLAZO:${diasOtorgados} DIAS | FECHA:${fechaActual}`;
                
                new QRCode(qrContainer, {
                    text: datosQR,
                    width: 110,
                    height: 110,
                    colorDark : "#047857", // Verde esmeralda (combina con la tarjeta)
                    colorLight : "#ffffff",
                    correctLevel : QRCode.CorrectLevel.H // Alta corrección de errores
                });

                // Mostrar el resultado con animación suave
                resultado.classList.remove('hidden');
                setTimeout(() => {
                    resultado.classList.remove('scale-95', 'opacity-0');
                    resultado.classList.add('scale-100', 'opacity-100');
                }, 10);

            }, 800);
        }

        // Función auxiliar para mostrar el error limpiamente
        function mostrarError(mensaje) {
            document.getElementById('textoError').innerText = mensaje;
            document.getElementById('mensajeError').classList.remove('hidden');
        }
    </script>
